def generate_value(key, type)
  case type
  when 'String'
    generate_string_value(key)
  when 'Integer'
    generate_integer_value(key)
  when 'Float'
    generate_float_value
  when 'Time'
    generate_time_value
  when 'Array<String>'
    generate_string_array_value
  else
    generate_default_value(key)
  end
end
