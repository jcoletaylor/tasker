def build(task, workflow_steps)
  edges = []

  workflow_steps.each do |step|
    next unless step.step_dag_relationship&.is_root_step

    edges << build_edge(
      "task_#{task.task_id}",
      "step_#{step.workflow_step_id}"
    )
  end

  edges
end
