def to_mermaid
  # For labeled edges, use the |label| syntax instead of adding a separate text label
  if label&.present? # rubocop:disable Lint/RedundantSafeNavigation
    "#{source_id} -->|\"#{escape_mermaid_text(label)}\"| #{target_id}"
  else
    "#{source_id} --> #{target_id}"
  end
end
