def initialize
  # Authentication defaults (assume no auth)
  @strategy = :none
  @options = {}
  @current_user_method = :current_user
  @authenticate_user_method = :authenticate_user!

  # Authorization defaults
  @coordinator_class = 'Tasker::Authorization::BaseCoordinator'
  @user_class = nil
  @enabled = false
end
